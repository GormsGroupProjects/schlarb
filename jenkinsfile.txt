pipeline{
    agent any

    stages{
        stage('Build'){
            steps{
                echo 'Building...'
                echo "Running ${env.BUILD_ID} ${env.BUILD_DISPLAY_NAME} on ${env.NODE_NAME} and JOB ${env.JOB_NAME}"
                //sh 'source .bashrc'
                echo "workspace: ${env.WORKSPACE}, JENKINS_HOME: ${env.JENKINS_HOME}"
                echo "ENV TEST: ${env.USER} ${env.RDS_USERNAME} ${env.RDS_PASSWORD}"
                sh '/var/lib/jenkins/workspace/Buttfuck/gradlew build'
            }
        }
        stage('Test'){
            steps{
                echo 'Testing...'
            }
        }
        stage('Deploy'){
            steps {
                echo 'Deploying...'
                sh '/var/lib/jenkins/workspace/Buttfuck/gradlew runJar'
            }
        }
    }
}
