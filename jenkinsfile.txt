pipeline{
    agent any

    stages{
        stage('Build'){
            steps{
                echo 'Building...'
                sh 'echo WORKSPACE: $WORKSPACE'
                echo "Running ${env.BUILD_ID} ${env.BUILD_DISPLAY_NAME} on ${env.NODE_NAME} and JOB ${env.JOB_NAME}"
                sh '/var/lib/jenkins/workspace/Buttfuck/gradlew build'
            }
        }
        stage('Test'){
            steps{
                echo 'No tests yet'
            }
        }
        stage('Test coverage') {
            steps {
                script {
                 try {
                        //sh 'chmod +x gradlew'
                        sh './gradlew test jacocoTestReport'
                    } finally {
                        junit '**/build/test-results/test/*.xml' 
                        //make the junit test results available in any case (success & failure)
                    }
                }
            }
        }
        stage('Deploy'){
            steps {
                echo 'Deploying...'
                sh '/var/lib/jenkins/workspace/Buttfuck/gradlew runJar'
            }
        }
    }
}
